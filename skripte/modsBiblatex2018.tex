%% Anzeige des Jahres nach dem Stichwort (usera) im Literaturverzeichnis
%% Wenn das Jahr bei Online-Quellen nicht explizit angegeben wurde, wird nach
%% dem Stichwort 'o. J.' ausgegeben. Nach der URL steht dann 'keine
%% Datumsangabe'. Ist das Jahr definiert, wird es an beiden Stellen ausgegeben.
%% Das Zugriffsdatum (urldate) spielt hier keine Rolle.
%% Für Nicht-Online-Quellen wird nichts geändert.
\renewbibmacro*{date+extradate}{%
  \printtext[parens]{%
    \printfield{usera}%
    \setunit{\printdelim{titleyeardelim}}%
    \ifentrytype{online}
    {\setunit*{\addspace\addcomma\addspace}%
      \iffieldundef{year}
      {\bibstring{nodate}}
      {\printlabeldateextra}}%
    {\printlabeldateextra}}}

% Anpassung für die Autorenformatierung
% Leitfaden IT-Management Anhang 2
% Nachname Autor1, Vorname Autor1, Nachname Autor2, Vorname Autor2, & Nachname Autor3, Vorname Autor3

\DeclareNameFormat{authorfirstfull}{%
  \ifnum\value{listcount}=1
    {\ifdefvoid{\namepartgiven}% Check, if first name is empty
    {\namepartfamily}% Only last name if no first name (e.g. company name)
    {\namepartfamily, \namepartgiven}}% Last name, full first name for the first author
  \else
    {\ifdefvoid{\namepartgiven} % Check, if first name is empty
      {\namepartfamily}% Only last name if no first name (e.g. company name)
      {\namepartfamily, \namepartgiveni}}% Last name, first initial for subsequent authors
  \fi
  \ifthenelse{\value{listcount}<\value{liststop}}{\addcomma\space}{}% Comma and space between authors
}

\DeclareNameAlias{author}{authorfirstfull}
